```python
def duplicated(words):
    # 결과 통
    result = []
    # 하나씩 확인하면서,
    for word in words:
    	# 중복을 확인하고, 결과에 없으면 넣는다
```

```python
def snail(height, day, night):
    # 날짜 지난 것 기록
    count = 0
    while True:
        count += 1
        # 낮, 가야할 높이가 낮아짐
        height -= day
        if height <= 0:
            return count
        # 밤, 가야할 높이가 높아짐
        height += night
```

### 편-안한 단어

```python
def comfortable_word(word):
    # 왼손 문자 목록, 오른손 문자 목록
    left_word, right_word = 'qwertasdfgzxcvb', 'yuiophjklnm'
    # 하나씩 반복하는데, 필요한 것은 이전 문자의 왼쪽인지 오른쪽인지 여부
    is_left = True if word[0] in left_word else False # 초기값을 지정
    for char in word[1:]: # 두번째부터 확인
        # 이전 것이 왼쪽이고, 지금 단어가 왼쪽이면 종료(False)
        if is_left and char in left_word:
            return False
        # 오른쪽 - 오른쪽이면 종료(False)
        if not is_left and char in right_word:
            return False
        # 맞으면, 왼-오 바꾸기
        is_left = not is_left
    # 종료 안되었으면 True
    return True
```

```python
def my_swapcase(word):
    # +32 대문자 -> 소문자
    # -32 소문자 -> 대문자
    result = ''
    for char in word:
        # 만약에 대문자라면
        if 65 <= ord(char) <= 90:
            result += chr(ord(char) + 32)
        # 만약에 소문자라면
        elif 97 <= ord(char) <=122:
            result += chr(ord(char) - 32)
        # 알파벳이 아닐 때
        else:
            result += char
    return result
```

```python
def unused_digits(*numbers):
    result = set('1234567890')
    for num in numbers:
        result -= set(str(num))
    return ''.join(sorted(result))
```

```python
    # 결과통
```
