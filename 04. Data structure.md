### abs()

```python
def my_abs(number):
    # 복소수일 때,
    if isinstance(number,complex):
        return (number.imag**2 + number.real**2)**0.5
    # 실수 및 정수일 때,
    if number == 0:
        return number**2
    elif number > 0:
        return number
    else:
        return -number
```



### 문자열 탐색

```python
def start_end(words):
    count = 0
    # 리스트를 반복하면서
    for word in words:
    	# 만약에 len(문자열) >=2 and 첫번째/마지막 같은지
    	if len(word) >= 2 and word[0] == word[-1]:
            count += 1
    # 해당하는 문자열 개수 리턴
    return count
```

```python
def start_end(words):
    return len([1 for word in words if len(word) >= 2 and word[0] == word[-1]])
```



### Collatz 추측

```python
def collatz(number):
    # 500번 시행
    for i in range(500):
        # 홀수
        if number % 2:
            number = number*3 + 1
        # 짝수
        else:
            number = number / 2
        # 1이 되면 종료(return)
        if number == 1:
            return i + 1 # 0~499니까 +1해서 반환
   return -1
```

### 무엇이 중복일까

```python
def duplicated(words):
    # 결과 통
    result = []
    # 하나씩 확인하면서,
    for word in words:
    	# 중복을 확인하고, 결과에 없으면 넣는다
        if words.count(word) > 1 and word not in result:
            result.append(word)
    return result
```

```python
def duplicated(words):
    return list({word for word in words if words.count(word) > 1})
```
