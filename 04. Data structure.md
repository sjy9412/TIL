```python
def my_abs(number):
    # 복소수일 때,
    if isinstance(number,complex):
        return (number.imag**2 + number.real**2)**0.5
    # 실수 및 정수일 때,
    if number == 0:
        return number**2
    elif number > 0:
        return number
    else:
        return -number
```

### 문자열 탐색

```python
def start_end(words):
    count = 0
    # 리스트를 반복하면서
    for word in words:
    	# 만약에 len(문자열) >=2 and 첫번째/마지막 같은지
    	if len(word) >= 2 and word[0] == word[-1]:
            count += 1
    # 해당하는 문자열 개수 리턴
    return count
```

```python
def start_end(words):
    return len([1 for word in words if len(word) >= 2 and word[0] == word[-1]])
```

### 무엇이 중복일까

```python
def duplicated(words):
    # 결과 통
    result = []
    # 하나씩 확인하면서,
    for word in words:
    	# 중복을 확인하고, 결과에 없으면 넣는다
        if words.count(word) > 1 and word not in result:
            result.append(word)
    return result
```

```python
def duplicated(words):
    return list({word for word in words if words.count(word) > 1})
```

### 불쌍한 달팽이

```python
def snail(height, day, night):
    # 날짜 지난 것 기록
    count = 0
    while True:
        count += 1
        # 낮, 가야할 높이가 낮아짐
        height -= day
        if height <= 0:
            return count
        # 밤, 가야할 높이가 높아짐
        height += night
```



### 편-안한 단어

```python
def comfortable_word(word):
    # 왼손 문자 목록, 오른손 문자 목록
    left_word, right_word = 'qwertasdfgzxcvb', 'yuiophjklnm'
    # 하나씩 반복하는데, 필요한 것은 이전 문자의 왼쪽인지 오른쪽인지 여부
    is_left = True if word[0] in left_word else False # 초기값을 지정
    for char in word[1:]: # 두번째부터 확인
        # 이전 것이 왼쪽이고, 지금 단어가 왼쪽이면 종료(False)
        if is_left and char in left_word:
            return False
        # 오른쪽 - 오른쪽이면 종료(False)
        if not is_left and char in right_word:
            return False
        # 맞으면, 왼-오 바꾸기
        is_left = not is_left
    # 종료 안되었으면 True
    return True
```



### 숫자패턴

```python
def pattern(number):
    # 문자열 통
    result = ''
    if number <= 0:
        return result
    # 1 ~ 해당숫자까지 반복
    for i in range(1, number+1):
        # 홀수면
        if i % 2:
            result += str(i)*i + '\n'
        	if i != number:
                result += '\n'
    return result
```

```python
def pattern(number):
    return '\n'.join([str(i)*i for i in range(1, number+1) if i % 2]) if number > 0 else ''
```

```python
def pattern(number):
    return '\n'.join([str(i)*i for i in range(1, number+1, 2)]) if number > 0 else ''
```
